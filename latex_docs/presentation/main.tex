def load_profiles(self):
        file = open("data/json/profiles.json")
        data = json.load(file)
        file.close()
        idx = 0
        for profile in data:
            new_profile = Profile(profile['name'], idx ,profile["port"]
                                  ,profile["remote"],profile["protocol"],profile["key"])
            self.profiles.append(new_profile) 
            idx += 1
\end{lstlisting}
